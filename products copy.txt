        async function fetchDataAndDisplay() {
            // Select the <ul> list
            const ulList = document.querySelector('.disha-image-list');
            // Clear existing <li> items
            ulList.innerHTML = '';

            // Fetch data from Firestore
            const querySnapshot = await getDocs(collection(db, "products"));

            const imageUrlsToCache = [];

            querySnapshot.forEach(async (doc) => {
                const data = doc.data();
                const imageUrl = await getDownloadURL(ref(storage, data.image));
                imageUrlsToCache.push(imageUrl);

                // Create <li> element
                const li = document.createElement('li');
                li.className = 'disha-image-item text-center';
                li.setAttribute('data-bottom-top', 'transform: translateY(40px)');
                li.setAttribute('data-top-bottom', 'transform: translateY(-40px)');





                li.innerHTML = `
        <div class="disha-shimmer-effect"></div>
        <a href="products_single_page.html?documentId=${doc.id}"><div class="disha-image-container">
            <img loading="lazy" src="${imageUrl}" alt="${data.products_name_one}" class="disha-image">
        </div> </a>
        
        
<div class="portfolio-caption pt-35px pb-35px md-pt-25px md-pb-25px">

                                    <span
                                        class="d-inline-block align-middle w-10px separator-line-1px bg-light-gray ms-10px me-10px xl-ms-5px xl-me-5px"></span>
                                    <div class="d-inline-block">${data.products_name_two}</div>
                                </div>
    `;

                // Append <li> to <ul>
                ulList.appendChild(li);

                // Preload and replace shimmer effect with image
                const img = new Image();
                img.src = imageUrl;
                img.onload = () => {
                    const shimmerEffect = li.querySelector('.disha-shimmer-effect');
                    const imageContainer = li.querySelector('.disha-image-container');

                    // Replace shimmer effect with image
                    imageContainer.innerHTML = `<img src="${imageUrl}" alt="${data.products_name_one}" class="disha-image">`;
                    shimmerEffect.style.display = 'none';
                };
            });


        }






        // Call the function to fetch data and display
        fetchDataAndDisplay().catch((error) => {
            console.log("Error fetching data: ", error);
        });
