        // Function to fetch data from Firestore and dynamically create <li> elements
        function fetchDataAndRender() {
            var serviceList = document.getElementById('service-list');

            // Reference to the collection you want to fetch data from
            var collectionRef = firestore.collection('services');

            // Fetch documents from the collection
            collectionRef.get().then(function (querySnapshot) {
                querySnapshot.forEach(function (doc) {
                    var data = doc.data();
                    // Create <li> element
                    var li = document.createElement('li');
                    li.className = 'grid-item selected digital transition-inner-all';
                    li.innerHTML = `

                        <div class="portfolio-box" data-bottom-top="transform: translateY(-20px)"
                                data-top-bottom="transform: translateY(-80px)">
                                <div class="portfolio-image bg-medium-gray border-radius-6px">
                                    <a href="demo-web-agency-single-project-creative.html"><img
                                            src="images/assets/solution.png" alt="" /></a>
                                </div>
                                <div class="portfolio-caption pt-35px pb-35px md-pt-25px md-pb-25px">
                                    <a href="demo-web-agency-single-project-creative.html"
                                        class="text-dark-gray text-dark-gray-hover fw-600">${data.service_name_one}</a>
                                    <span
                                        class="d-inline-block align-middle w-10px separator-line-1px bg-light-gray ms-10px me-10px"></span>
                                    <div class="d-inline-block">Solutions</div>
                                </div>
                            </div>
                    `;
                    // Append <li> to the service list
                    serviceList.appendChild(li);
                });
            }).catch(function (error) {
                console.log("Error getting documents: ", error);
            });
        }

        // Call the function to fetch data and render on page load
        fetchDataAndRender();
